## Tâches

### 1. Charge ta configuration et l'historique des positions
```python
import json, os
from datetime import datetime, timezone, timedelta

cfg_path = f"{PROJECT_DIR}/config.json"
trade_hist_path = f"{PROJECT_DIR}/state/trade_history.json"

with open(cfg_path) as f:
    cfg = json.load(f)

with open(trade_hist_path) as f:
    trade_history = json.load(f)

min_profit_pct = cfg.get("min_profit_pct_take", 2.0)
max_hold_days = cfg.get("max_hold_days", 14)

positions = []
for entry in trade_history:
    if entry.get("status") == "open":
        positions.append(entry)
```

### 2. Récupère les ordres actifs (bot + manuels) et prix actuels
```python
# Ordres bot (de trade_history.json)
open_orders_bot = {}
for pos in positions:
    coin = pos["coin"]
    open_orders_bot[coin] = pos

# Ordres manuels (via kraken open-orders)
open_orders_raw = json.loads(binance("open-orders", "-o", "json")) if True else {}
all_open = open_orders_raw.get("open", {})

manual_positions = {}
for txid, order in all_open.items():
    pair = order.get("descr", {}).get("pair", "")
    if pair.endswith("USDC"):
        coin = pair.replace("USDC", "")
        order["_txid"] = txid  # conservé pour annulation éventuelle
        manual_positions[coin] = order

# Merge : tous les coins (bot + manuels)
all_coins = set(open_orders_bot.keys()) | set(manual_positions.keys())

# Récupère prix actuels pour tous les coins
current_prices = {}
for coin in all_coins:
    try:
        price_json = binance("ticker", f"{coin}USDC", "-o", "json")
        price_data = json.loads(price_json)
        current_prices[coin] = float(price_data.get(f"{coin}USDC", {}).get("c", [0])[0])
    except:
        current_prices[coin] = 0
```

### 3. Recalibrage TP via résistances TradingView
```python
TV_MAP = {"XBT": "BTC", "XDG": "DOGE"}
reward_risk_ratio = cfg.get("reward_risk_ratio", 2.0)
changed = False

for pos in positions:
    coin = pos["coin"]
    tv_symbol = TV_MAP.get(coin, coin)
    entry_price = float(pos.get("entry_price", 0))
    stop_price = float(pos.get("stop_price", 0))
    tp_actuel = float(pos.get("tp_price", 0))

    if entry_price <= 0 or stop_price <= 0 or tp_actuel <= 0:
        continue

    try:
        analysis = mcp__tradingview__combined_analysis(symbol=f"{tv_symbol}USDT", exchange="BINANCE", timeframe="4h")
        r2_4h = analysis.get("technical", {}).get("support_resistance", {}).get("resistance_2")
    except:
        continue

    # Assumes long position (stop < entry); short positions would produce negative distance and invalid tp_mecanique
    assert stop_price < entry_price, f"stop_price {stop_price} must be < entry_price {entry_price} for long position (coin {coin})"
    stop_distance_pct = (entry_price - stop_price) / entry_price
    tp_mecanique = entry_price * (1 + stop_distance_pct * reward_risk_ratio)

    if r2_4h and float(r2_4h) > entry_price:
        tp_smart = min(tp_mecanique, float(r2_4h) * 0.98)
        source = f"R {float(r2_4h):.6f} × 0.98"
    else:
        tp_smart = tp_mecanique
        source = "mécanique"

    if abs(tp_smart - tp_actuel) / tp_actuel > 0.005:
        pos["tp_price"] = tp_smart
        changed = True
        tg(f"📐 TP {coin} recalibré : {tp_actuel:.6f} → {tp_smart:.6f} ({source})")

if changed:
    _save_trade_history_atomic(trade_history)
```

### 4. Évalue chaque position (bot et manuelles)
```python
actions = []
now = datetime.now(timezone.utc)

for coin in all_coins:
    pos_bot = open_orders_bot.get(coin)
    pos_manual = manual_positions.get(coin)
    current_price = current_prices.get(coin, 0)

    if not current_price:
        continue

    # Positions bot : utilise entry_price et entry_time
    if pos_bot:
        entry_price = float(pos_bot.get("entry_price", 0))
        qty = float(pos_bot.get("quantity", 0))
        entry_time_str = pos_bot.get("date", "")

        if not entry_price:
            continue

        pnl_pct = ((current_price - entry_price) / entry_price) * 100

        # Durée depuis l'entrée
        try:
            entry_time = datetime.fromisoformat(entry_time_str.replace("Z", "+00:00"))
            hold_days = (now - entry_time).days
        except:
            hold_days = 0

        # Critère 1: Profit seuil atteint
        if pnl_pct >= min_profit_pct:
            reason = f"Profit {pnl_pct:.2f}% >= seuil {min_profit_pct}% (bot)"
            actions.append({"coin": coin, "qty": qty, "reason": reason, "source": "bot"})

        # Critère 2: Position en perte depuis max_hold_days
        elif pnl_pct < 0 and hold_days >= max_hold_days:
            reason = f"Perte {pnl_pct:.2f}%, durée {hold_days}j >= {max_hold_days}j (bot)"
            actions.append({"coin": coin, "qty": qty, "reason": reason, "cut_loss": True, "source": "bot"})

    # Positions manuelles (OCO) : calcule P&L depuis le prix d'entrée du BUY
    if pos_manual and not pos_bot:
        orders = pos_manual.get("orders", [])
        buy_order = next((o for o in orders if o.get("side") == "BUY"), None)

        if buy_order:
            entry_price = float(buy_order.get("price", 0))
            qty = float(buy_order.get("origQty", 0))

            if entry_price > 0:
                pnl_pct = ((current_price - entry_price) / entry_price) * 100

                # OCO manuel : si P&L >= seuil, annuler OCO et vendre
                if pnl_pct >= min_profit_pct:
                    reason = f"Profit {pnl_pct:.2f}% >= seuil {min_profit_pct}% (OCO manuel)"
                    actions.append({"coin": coin, "qty": qty, "reason": reason, "source": "manual", "cancel_oco": True})

                # Sinon, laisser Binance gérer l'OCO actif
```

### 5. Exécute les ventes (et annule OCO si manuel)
```python
for action in actions:
    coin = action["coin"]
    qty = action["qty"]
    reason = action["reason"]
    source = action.get("source", "unknown")
    cancel_oco = action.get("cancel_oco", False)

    # Si ordre manuel Kraken : annuler le SL avant SELL MARKET
    if cancel_oco:
        try:
            pos_manual = manual_positions.get(coin)
            if pos_manual:
                txid = pos_manual.get("_txid")
                if txid:
                    try:
                        binance("order", "cancel", txid, "-o", "json", "--yes")
                    except:
                        pass
        except Exception as e:
            tg(f"⚠️ Annulation ordre {coin} (partielle): {e}")

    # Vendre au marché
    sell_cmd = ["market-order", "sell", coin, "USDC", str(qty)]
    try:
        sell_result = binance(*sell_cmd)
        tg(f"✅ Position {coin}: Vendu {qty} ({reason})")
    except Exception as e:
        tg(f"❌ Vente {coin} échouée: {e}")
```
